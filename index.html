<body>

    <div id="weather">

        Loading weather...

    </div>

    

    <script>

        const apiKey = ‘f0cdca0ce016f8d05cab47fa156f375a’

        const city = ‘’Victoria’; // Replace with your desired city

        const units = 'metric'; // 'metric' for Celsius, 'imperial' for Fahrenheit

        const weatherDiv = document.getElementById('weather');


        function getWeather() {

            fetch(`https://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}&units=${units}`)

                .then(response => {

                    if (!response.ok) {

                        throw new Error('City not found');

                    }

                    return response.json();

                })

                .then(data => {

                    const temperature = data.main.temp;

                    const weatherDescription = data.weather[0].description;

                    weatherDiv.innerHTML = `Temperature: ${temperature}°C<br>Description: ${weatherDescription}`;

                })

                .catch(error => {

                    weatherDiv.innerHTML = 'Error retrieving weather data.';

                    console.error(error);

                });

        }


        getWeather(); // Retrieve weather data when the page loads

    </script>
